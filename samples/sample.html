Code Sample

-- lua/void-space/palette.lua
local M = {}

M.void    = "#0d0f18"  -- deep space background
M.nebula  = "#c792ea"  -- purple — keywords & types
M.aurora  = "#89ddff"  -- cyan   — operators & special
M.comet   = "#82aaff"  -- blue   — functions & modules
M.pulsar  = "#ff5370"  -- red    — errors & deleted
M.quasar  = "#ffcb6b"  -- yellow — strings & warnings

return M
